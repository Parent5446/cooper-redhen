00024 ## Search for a spectrum.
00025 # 
00026 # Parse the given file and create a Spectrum object for it. Use the Matcher
00027 # object to find candidates for similar spectra in the database and compare
00028 # all candidates to the original spectrum using linear comparison algorithms.
00029 #
00030 # @param file File descriptor for the spectrum
00031 # @return List of Spectrum objects
00032 def search(file_obj):
00033     """Search for a spectrum based on a given file descriptor."""
00034     # Load the user's spectrum into a Spectrum object.
00035     spectrum = Spectrum()
00036     spectrum.parse_file(file_obj)
00037     # Check cache for the Matcher. If not, get from database.
00038     matcher = memcache.get(spectrum.type + '_matcher')
00039     if matcher is None:
00040         matcher = Matcher.get_by_key_name(spectrum.type)
00041     # Get the candidates for similar spectra.
00042     candidates = matcher.get(spectrum)
00043     # Do one-to-one on candidates and sort by error
00044     candidates = [Matcher.bove(spectrum, candidate) for candidate in candidates]
00045     list.sort(candidates, key=operator.attrgetter('error'))
00046     # Let frontend do the rest
00047     return candidates

00094     # Search a JCAMP file for the chemical's name, type, and spectrum data.

00098     # @param file_obj File descriptor for the JCAMP file.
00099     def parse_file(self, file_obj):
00100         """Parse a string of JCAMP file data and extract all needed data."""
00101         self.contents = file_obj.read()
00102         self.type = 'Infrared' # Later this will be variable
00103         x = float(self.get_field('##FIRSTX=')) # The first x-value
00104         deltaX = float(self.get_field('##DELTAX=')) # The Space between adjacent x values
00105         xFactor = float(self.get_field('##XFACTOR=')) # for our purposes it's 1, but if not use this instead
00106         yFactor = float(self.get_field('##YFACTOR=')) # some very small number, but if not use this instead
00107         self.xy = []
00108         # Process the XY data from JCAMP's (X++(Y..Y)) format.
00109         raw_xy = self.contents[self.contents.index('##XYDATA=(X++(Y..Y))') + 20:]
00110         for match in re.finditer(r'(\D)([\d.-]+)', raw_xy):
00111             if match.group(1) == '\n':
00112                 # Number is the first on the line and is an x-value
00113                 x = float(match.group(2)) * xFactor
00114             else:
00115                 # Number is a relative y-value.
00116                 self.xy.append((x, float(match.group(2))*yFactor))
00117                 x += deltaX
00118         # Keep the data in ascending order. It will be descending in the file
00119         # if our delta X is negative.
00120         if deltaX < 0:
00121             self.xy.reverse()
00122         # Integrate self.xy numerically over a fixed range.
00123         range = (700.0, 3900.0)
00124         # Initialize the data and find the interval of integration.
00125         self.data = [0.0 for i in xrange(1000)]
00126         interval = (range[1] - range[0]) / len(self.data)
00127         # Find index in self.xy where integrals start
00128         start = bisect.bisect_left(self.xy, (range[0], 0))
00129         # oldX = start of range, oldY = linear interpolation of corresponding y
00130         xy = self.xy
00131         oldX, oldY = range[0], (self.xy[start - 1][1] +
00132              (xy[start][1] - xy[start - 1][1]) * (range[0] - xy[start][0]) /
00133              (xy[start - 1][0] - xy[start][0]))
00134         for x, y in self.xy[start:]: #Iterate over self.xy from start
00135             newIndex = int((x - range[0]) / interval)
00136             oldIndex = int((oldX - range[0]) / interval)
00137             if newIndex != oldIndex:
00138                 # We're starting a new integral.
00139                 boundary = newIndex * interval,\
00140                            ((y - oldY) * (newIndex * interval - oldX) /
00141                            (x - oldX) + oldY) #Linear interpolation
00142                 self.data[oldIndex] += (boundary[1] + oldY) * (boundary[0] - oldX) / 2 #Add area
00143                 if newIndex < len(self.data): # if data isn't filled 
00144                     self.data[newIndex] += (boundary[1] + y) * (x - boundary[0]) / 2 #Add area
00145             else:
00146                 self.data[newIndex] += (y + oldY) * (x - oldX) / 2 #Add area
00147             if x > range[1]:
00148                 break #If finished, break
00149             oldX, oldY = x, y #Otherwise keep going
00150         self.chemical_type = 'Unknown' # We will find this later
00151         # FIXME: Assumes chemical name is in TITLE label.
00152         self.chemical_name = self.get_field('##TITLE=')
00153         # Reference: http://www.jcamp-dx.org/

00293         index, peaks, and other indices and add them to the data structures."""

00295         
00296         #Flat heavyside: hash table of heavyside keys
00297         stack = [(0,0,0,len(spectrum.data))] #List of (key, whichBit, leftEdge, width)
00298         while len(stack) > 0: #keep going while there is stuff on the stack
00299             key, whichBit, leftEdge, width = stack.pop() #stack holds key, whichBit, leftEdge, width
00300             if whichBit == Matcher.FLAT_HEAVYSIDE_BITS: #We're done with this key, so add it to the table
00301                 if key in self.flat_heavyside: self.flat_heavyside[key].add(spectrum.key()) #add it to table
00302                 else: self.flat_heavyside[key] = set([spectrum.key()]) #add it to table
00303             else:
00304                 left = sum(spectrum.data[leftEdge:leftEdge+width/2]) #Sum integrals
00305                 right = sum(spectrum.data[leftEdge+width/2:leftEdge+width]) #on both sides
00306                 if leftEdge+width == len(spectrum.data): leftEdge = 0; width = width/2 #Adjust boundaries
00307                 else: leftEdge += width #for next iteration
00308                 if abs(left-right) < left*0.03: #If too close to call add twice
00309                     stack.append( (key, whichBit+1, leftEdge, width) ) #Once, leave key unchanged
00310                     stack.append( (key+(1<<(Matcher.FLAT_HEAVYSIDE_BITS-whichBit)), whichBit+1, leftEdge, width) ) #and once change key
00311                 else: stack.append( (key+((left<right)<<(Matcher.FLAT_HEAVYSIDE_BITS-whichBit)), whichBit+1, leftEdge, width) ) #if its easy to tell which side is heavier

00314         xy = sorted(spectrum.xy, key = operator.itemgetter(1), reverse = True) #sort xy data by y value
00315         peaks = [x for x, y in xy if y >= xy[0][1]*0.95 and not [peak for peak in peaks if abs(peak - x) < 1]]
